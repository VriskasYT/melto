// Streaming chat с Melto AI const response = await fetch("https://YOUR-APP.replit.app/api/chat/send", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ sessionId: crypto.randomUUID(), message: "Напиши функцию на JavaScript" }) }); const reader = response.body.getReader(); const dec = new TextDecoder(); while (true) { const { done, value } = await reader.read(); if (done) break; for (const line of dec.decode(value).split("\n")) { if (line.startsWith("data: ")) { const d = JSON.parse(line.slice(6)); if (d.chunk) process.stdout.write(d.chunk); } } }
import requests, json, uuid session_id = str(uuid.uuid4()) with requests.post( "https://YOUR-APP.replit.app/api/chat/send", json={"sessionId": session_id, "message": "Привет!"}, stream=True ) as r: for line in r.iter_lines(): if line.startswith(b"data: "): d = json.loads(line[6:]) if "chunk" in d: print(d["chunk"], end="", flush=True)
